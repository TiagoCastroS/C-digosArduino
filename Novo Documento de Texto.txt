void setup() {
  pinMode(ama, OUTPUT);
  pinMode(sab, OUTPUT);
  pinMode(bomb_fre, OUTPUT);
  pinMode(hor, OUTPUT);
  pinMode(ant, OUTPUT);
  pinMode(pres, INPUT);
  pinMode(tam, INPUT);

  int est_tam = digitalRead(tam);
  if (est_tam == LOW) {
    digitalWrite (ama, HIGH);
    digitalWrite (sab, HIGH);
    digitalWrite (bomb_fre, HIGH);
    digitalWrite (hor, HIGH);
    digitalWrite (ant, HIGH);
  }

  lcd.begin(16, 2);
  Serial.begin(9600);
  digitalWrite (ama, HIGH);
  digitalWrite (sab, HIGH);
  digitalWrite (bomb_fre, HIGH);
  digitalWrite (hor, HIGH);
  digitalWrite (ant, HIGH);
  lcd.setCursor(0, 0);
  lcd.print("BEM VINDO A SUA");
  lcd.setCursor(0, 1);
  lcd.print("MAQUINA DE LAVAR");
  delay(3000);
  lcd.clear();
  lcd.setCursor(2, 0);
  lcd.print("ESCOLHA SEU");
  lcd.setCursor(0, 1);
  lcd.print("MODO DE LAVAGEM");
  delay(3000);
  lcd.clear();
}

void (*reset) () = 0;

boolean bater() {
  indice = 2;
  for (maxi = 0; maxi < indice; maxi++) {

    digitalWrite(hor, LOW);
    delay(1000);
    digitalWrite(hor, HIGH);
    delay(2000);
    digitalWrite(ant, LOW);
    delay(1000);
    digitalWrite(ant, HIGH);
    delay(1000);
  }


  return true;
}

boolean encherum() {

  }

    return false;
  }
}

boolean esperar(long tempo) {
  delay(tempo);
  return true;
}
boolean esvaziar() {
  delay(2000);
  digitalWrite(bomb_fre, LOW);
  delay(30000); // tempo que se imagina que a máquina vá estar vazia
  digitalWrite(bomb_fre, HIGH);

  return true;
}

boolean enxague() {
  esperar(1500);
  indice = 2;
  for (maxi = 0; maxi < indice; maxi++) {

    digitalWrite(hor, LOW);
    delay(1000);
    digitalWrite(hor, HIGH);
    delay(2000);
    digitalWrite(ant, LOW);
    delay(1000);
    digitalWrite(ant, HIGH);
    delay(1000);
  }
  delay(2000);
  digitalWrite(bomb_fre, LOW);
  delay(30000); // tempo que se imagina que a máquina vá estar vazia
  digitalWrite(bomb_fre, HIGH);

  return true;
}

boolean centrifugar() {
  digitalWrite (bomb_fre, LOW);
  delay(4000);
  digitalWrite (hor, LOW);
  esperar(15000); // tempo imaginário para centrifugar
  digitalWrite (bomb_fre, HIGH);
  digitalWrite (hor, HIGH);

  return true;
}

boolean autolimpeza () {
  if (fim_bater == false) {
    confirma_Encher = encherum();
  }

  if (confirma_Encher == true && fim_bater == false) {
    confirma_bater = bater();
    fim_bater = true;
  }
  if (confirma_bater == true && fim_bater == true && confirma_esvaziar == false) {
    confirma_esvaziar = esvaziar();
  }
}

boolean mt_suja() {
  if (fim_da_lavagem == false) {
    confirma_Encher = encherum();
    est = false;
  }

  if (confirma_Encher == true && fim_da_lavagem == false && fim_encherum == true ) {
    confirma_turbo = turbo();
    molholon();
    confirma_esvaziar = esvaziar();
    fim_da_lavagem = true;
  }

  if (confirma_esvaziar == true && fim_da_lavagem == true && confirma_enxague == false) {
    confirma_encherdois = encherdois();
    est = false;
    fim = true;
  }

  if (confirma_encherdois == true &&  fim_da_lavagem == true && fim == true && fim_centrifuga == false && fim_sla == false ) {
    delay(1500);
    confirma_enxague = enxague();


    if (confirma_enxague == true && fim_sla == false) {
      Serial.println("enchendo pela segunda vez");
      delay(300);
      confirma_encherdois = encherdois();
      est = false;
      fim_sla = true;
    }
    if (confirma_encherdois == true && est == false  && fim_sla == true && fim_da_lavagem == true);
    Serial.println("segundo enxague");
    delay(300);
    confirma_enxague = enxague();
    fim_enxague = true;
  }


  if (confirma_enxague == true && fim_enxague == true) {
    confirma_centrifugar = centrifugar ();
    fim_centrifuga = true;
  }
}

void loop() {

  est = digitalRead(pres);
  est_limpeza = digitalRead(limpeza);

  int est_tam = digitalRead(tam);

  digitalWrite (ama, HIGH);
  digitalWrite (sab, HIGH);
  digitalWrite (bomb_fre, HIGH);
  digitalWrite (hor, HIGH);
  digitalWrite (ant, HIGH);


  if (botao < 100) {

    ok ++;
    if (ok > 15) {
      ok = 1;
    }

    if (ok == 1) {
      modo = "MUITO SUJA";
    }
    if (ok == 2) {

      modo = "UNIFORME";
    }

      modo = "COLORIDAS";

      modo = "POUCO SUJAS";

      modo = "INTIMAS";

      modo = "ENXAGUAR";
    }
    if (ok == 14) {
      modo = "CENTRIFUGAR";
    }
    if (ok == 15) {
      modo = "AUTO LIMPEZA";
    }


    lcd.clear();
    lcd.setCursor (0, 0);
    lcd.print("MODO: ");
    lcd.setCursor (0, 1);
    lcd.print(modo);
    delay(300);
  }

    OK_Select = true;

      lcd.print("LAVAGEM");

    }
  }
  if (ok == 11 && OK_Select == true) {

      lcd.print("LAVAGEM");

    }

  }
  if (ok == 15 && OK_Select == true) {
    autolimpeza();
    lcd.setCursor(0, 0);
    lcd.print("MODO ESPECIAL: ");
    lcd.setCursor(0, 1);
    lcd.print("AUTO LIMPEZA");
    autolimpeza();
    if (confirma_esvaziar == true) {
      lcd.clear();
      lcd.setCursor(1, 0);
      lcd.print("MODO ESPECIAL ");
      lcd.setCursor(4, 1);

      esperar(8000);
      reset();
    }
  }
